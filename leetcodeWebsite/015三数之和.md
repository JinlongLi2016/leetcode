# 015三数之和





* M1, time exceed

```python
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        s = set(nums)
        res = []
        i, j, k = 0, 1, 2
        n = len(nums)
        while i < n:
            
            j = i + 1
            while j < n:
                k = j + 1
                while k < n:
                    if nums[i] + nums[j] + nums[k] == 0:
                        t = [nums[i], nums[j], nums[k]]
                        t.sort()
                        if t not in res:
                            res.append(t)
                    k += 1
                j += 1
            i += 1
        return res
```
