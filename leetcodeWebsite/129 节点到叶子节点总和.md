# 129 节点到叶子节点总和



```python
def dfs(t, preS):
    global totalSum
    if not t:
        return 
    if not t.right and not t.left :
        totalSum += preS * 10 + t.val
        return 
    dfs(t.left, 10 * preS + t.val)
    dfs(t.right, 10 * preS + t.val)
    
class Solution:
    def sumNumbers(self, root: TreeNode) -> int:
        global totalSum
        totalSum = 0
        dfs(root, 0)
        return totalSum
```
